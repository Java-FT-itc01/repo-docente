# 03. LAB09 - Mini-progetto procedurale con menu e array

## Obiettivo del laboratorio

In questo laboratorio costruirai un mini-progetto procedurale da console.

Il programma permetterà di gestire un array di interi tramite un menu.

Il programma dovrà consentire di:

1. inserire o sostituire un array;
2. stampare l'array;
3. calcolare somma e media;
4. trovare minimo e massimo;
5. cercare un valore;
6. contare le occorrenze di un valore;
7. mostrare una copia ordinata crescente;
8. verificare se l'array è già ordinato;
9. uscire dal programma.

Questo laboratorio integra molti argomenti precedenti:

- input robusto;
- `Scanner`;
- `nextLine()`;
- `try/catch`;
- array;
- metodi statici;
- ricerca;
- conteggio;
- minimo;
- massimo;
- media;
- ordinamento;
- menu con `switch`;
- ciclo principale.

## 5. Utility di input

```java
public static int leggiIntero(Scanner scanner, String messaggio) {
    while (true) {
        System.out.print(messaggio);
        String testo = scanner.nextLine();

        try {
            return Integer.parseInt(testo);
        } catch (NumberFormatException e) {
            System.out.println("Errore: devi inserire un intero valido.");
        }
    }
}
```

```java
public static int leggiInteroPositivo(Scanner scanner, String messaggio) {
    while (true) {

        if (valore > 0) {
            return valore;
        }

        System.out.println("Errore: il valore deve essere maggiore di zero.");
    }
}
```

## 13. Ordinamento crescente su copia

Aggiungi:

```java
public static void ordinaCrescente(int[] array) {
    for (int i = 0; i < array.length - 1; i++) {
        for (int j = i + 1; j < array.length; j++) {
            if (array[i] > array[j]) {
                int temp = array[i];
                array[i] = array[j];
                array[j] = temp;
            }
        }
    }
}
```

Aggiungi:

```java
public static void stampaCopiaOrdinata(int[] array) {
    if (!arrayDisponibile(array)) {
        System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
        return;
    }

    int[] copia = Arrays.copyOf(array, array.length);
    ordinaCrescente(copia);

    System.out.println("Array originale: " + Arrays.toString(array));
    System.out.println("Copia ordinata: " + Arrays.toString(copia));
}
```

Usiamo una copia per non modificare l'array originale.

## 16. Codice finale completo

Confronta il tuo file con questa versione:

```java
import java.util.Scanner;
import java.util.Arrays;

public class MiniProgettoMenuArray {

    public static void stampaMenu() {
        System.out.println();
        System.out.println("=== MINI PROGETTO ARRAY ===");
        System.out.println("1. Inserisci o sostituisci array");
        System.out.println("2. Stampa array");
        System.out.println("3. Somma e media");
        System.out.println("4. Minimo e massimo");
        System.out.println("5. Cerca valore");
        System.out.println("6. Conta occorrenze");
        System.out.println("7. Ordina crescente");
        System.out.println("8. Verifica se array ordinato crescente");
        System.out.println("0. Esci");
    }

    public static int leggiIntero(Scanner scanner, String messaggio) {
        while (true) {
            System.out.print(messaggio);
            String testo = scanner.nextLine();

            try {
                return Integer.parseInt(testo);
            } catch (NumberFormatException e) {
                System.out.println("Errore: devi inserire un intero valido.");
            }
        }
    }

    public static int leggiInteroPositivo(Scanner scanner, String messaggio) {
        while (true) {
            int valore = leggiIntero(scanner, messaggio);

            if (valore > 0) {
                return valore;
            }

            System.out.println("Errore: il valore deve essere maggiore di zero.");
        }
    }

    public static int[] leggiArray(Scanner scanner, int dimensione) {
        int[] array = new int[dimensione];

        for (int i = 0; i < array.length; i++) {
            array[i] = leggiIntero(scanner, "Inserisci valore in posizione " + i + ": ");
        }

        return array;
    }

    public static int[] acquisisciArray(Scanner scanner) {
        int dimensione = leggiInteroPositivo(scanner, "Quanti numeri vuoi inserire? ");
        return leggiArray(scanner, dimensione);
    }

    public static boolean arrayDisponibile(int[] array) {
        return array != null;
    }

    public static void stampaArray(int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        System.out.println("Array: " + Arrays.toString(array));
    }

    public static int calcolaSomma(int[] array) {
        int somma = 0;

        for (int i = 0; i < array.length; i++) {
            somma += array[i];
        }

        return somma;
    }

    public static double calcolaMedia(int[] array) {
        return (double) calcolaSomma(array) / array.length;
    }

    public static void stampaSommaMedia(int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        System.out.println("Somma: " + calcolaSomma(array));
        System.out.println("Media: " + calcolaMedia(array));
    }

    public static int trovaMinimo(int[] array) {
        int minimo = array[0];

        for (int i = 1; i < array.length; i++) {
            if (array[i] < minimo) {
                minimo = array[i];
            }
        }

        return minimo;
    }

    public static int trovaMassimo(int[] array) {
        int massimo = array[0];

        for (int i = 1; i < array.length; i++) {
            if (array[i] > massimo) {
                massimo = array[i];
            }
        }

        return massimo;
    }

    public static void stampaMinimoMassimo(int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        System.out.println("Minimo: " + trovaMinimo(array));
        System.out.println("Massimo: " + trovaMassimo(array));
    }

    public static int cercaValore(int[] array, int valore) {
        for (int i = 0; i < array.length; i++) {
            if (array[i] == valore) {
                return i;
            }
        }

        return -1;
    }

    public static void eseguiRicerca(Scanner scanner, int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        int valore = leggiIntero(scanner, "Valore da cercare: ");
        int posizione = cercaValore(array, valore);

        if (posizione >= 0) {
            System.out.println("Valore trovato in posizione: " + posizione);
        } else {
            System.out.println("Valore non trovato.");
        }
    }

    public static int contaOccorrenze(int[] array, int valore) {
        int contatore = 0;

        for (int i = 0; i < array.length; i++) {
            if (array[i] == valore) {
                contatore++;
            }
        }

        return contatore;
    }

    public static void eseguiConteggio(Scanner scanner, int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        int valore = leggiIntero(scanner, "Valore da contare: ");
        int occorrenze = contaOccorrenze(array, valore);

        System.out.println("Occorrenze di " + valore + ": " + occorrenze);
    }

    public static void ordinaCrescente(int[] array) {
        for (int i = 0; i < array.length - 1; i++) {
            for (int j = i + 1; j < array.length; j++) {
                if (array[i] > array[j]) {
                    int temp = array[i];
                    array[i] = array[j];
                    array[j] = temp;
                }
            }
        }
    }

    public static void stampaCopiaOrdinata(int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        int[] copia = Arrays.copyOf(array, array.length);
        ordinaCrescente(copia);

        System.out.println("Array originale: " + Arrays.toString(array));
        System.out.println("Copia ordinata: " + Arrays.toString(copia));
    }

    public static boolean isOrdinatoCrescente(int[] array) {
        for (int i = 0; i < array.length - 1; i++) {
            if (array[i] > array[i + 1]) {
                return false;
            }
        }

        return true;
    }

    public static void stampaVerificaOrdinamento(int[] array) {
        if (!arrayDisponibile(array)) {
            System.out.println("Nessun array disponibile. Prima acquisisci i dati.");
            return;
        }

        if (isOrdinatoCrescente(array)) {
            System.out.println("L'array è ordinato in modo crescente.");
        } else {
            System.out.println("L'array non è ordinato in modo crescente.");
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        int[] numeri = null;
        int scelta;

        do {
            stampaMenu();
            scelta = leggiIntero(scanner, "Scelta: ");

            switch (scelta) {
                case 1:
                    numeri = acquisisciArray(scanner);
                    System.out.println("Array acquisito correttamente.");
                    break;
                case 2:
                    stampaArray(numeri);
                    break;
                case 3:
                    stampaSommaMedia(numeri);
                    break;
                case 4:
                    stampaMinimoMassimo(numeri);
                    break;
                case 5:
                    eseguiRicerca(scanner, numeri);
                    break;
                case 6:
                    eseguiConteggio(scanner, numeri);
                    break;
                case 7:
                    stampaCopiaOrdinata(numeri);
                    break;
                case 8:
                    stampaVerificaOrdinamento(numeri);
                    break;
                case 0:
                    System.out.println("Uscita dal programma.");
                    break;
                default:
                    System.out.println("Scelta non valida.");
            }
        } while (scelta != 0);

        scanner.close();
    }
}
```
