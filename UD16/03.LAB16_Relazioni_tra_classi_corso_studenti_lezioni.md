```java
package corso.lab16;

public class Studente {
    private String matricola;
    private String nome;
    private String cognome;

    public Studente(String matricola, String nome, String cognome) {
        setMatricola(matricola);
        setNome(nome);
        setCognome(cognome);
    }

    public String getMatricola() {
        return matricola;
    }

    public void setMatricola(String matricola) {
        if (matricola != null && !matricola.trim().isEmpty()) {
            this.matricola = matricola.trim();
        } else {
            this.matricola = "N/D";
        }
    }

    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        if (nome != null && !nome.trim().isEmpty()) {
            this.nome = nome.trim();
        } else {
            this.nome = "Nome non disponibile";
        }
    }

    public String getCognome() {
        return cognome;
    }

    public void setCognome(String cognome) {
        if (cognome != null && !cognome.trim().isEmpty()) {
            this.cognome = cognome.trim();
        } else {
            this.cognome = "Cognome non disponibile";
        }
    }

    @Override
    public String toString() {
        return "Studente{" 
                + "matricola='" + matricola + '\''
                + ", nome='" + nome + '\''
                + ", cognome='" + cognome + '\''
                + '}';
    }
}
```

`Studente` rappresenta un oggetto autonomo.

Uno studente può esistere anche fuori da un corso.

## 4. File `Corso.java`

Crea:

```text
src/corso/lab16/Corso.java
```

Inserisci:

```java
package corso.lab16;

import java.util.ArrayList;

public class Corso {
    private String codice;
    private String titolo;
    private ArrayList<Studente> studenti;
    private ArrayList<Lezione> lezioni;

    public Corso(String codice, String titolo) {
        setCodice(codice);
        setTitolo(titolo);
        studenti = new ArrayList<>();
        lezioni = new ArrayList<>();
    }

    public String getCodice() {
        return codice;
    }

    public void setCodice(String codice) {
        if (codice != null && !codice.trim().isEmpty()) {
            this.codice = codice.trim();
        } else {
            this.codice = "N/D";
        }
    }

    public String getTitolo() {
        return titolo;
    }

    public void setTitolo(String titolo) {
        if (titolo != null && !titolo.trim().isEmpty()) {
            this.titolo = titolo.trim();
        } else {
            this.titolo = "Corso senza titolo";
        }
    }

    public void iscriviStudente(Studente studente) {
        if (studente != null && !studenti.contains(studente)) {
            studenti.add(studente);
        }
    }

    public void aggiungiLezione(String titolo, int durataMinuti) {
        lezioni.add(new Lezione(titolo, durataMinuti));
    }

    public int getNumeroStudenti() {
        return studenti.size();
    }

    public int getNumeroLezioni() {
        return lezioni.size();
    }

    public int calcolaDurataTotaleMinuti() {
        int totale = 0;

        for (Lezione lezione : lezioni) {
            totale += lezione.getDurataMinuti();
        }

        return totale;
    }

    public void stampaStudenti() {
        System.out.println("Studenti iscritti al corso " + codice + ":");

        for (Studente studente : studenti) {
            System.out.println("- " + studente);
        }
    }

    public void stampaLezioni() {
        System.out.println("Lezioni del corso " + codice + ":");

        for (Lezione lezione : lezioni) {
            System.out.println("- " + lezione);
        }
    }

    public ArrayList<Studente> getStudenti() {
        return studenti;
    }

    @Override
    public String toString() {
        return "Corso{" 
                + "codice='" + codice + '\''
                + ", titolo='" + titolo + '\''
                + ", numeroStudenti=" + studenti.size()
                + ", numeroLezioni=" + lezioni.size()
                + ", durataTotaleMinuti=" + calcolaDurataTotaleMinuti()
                + '}';
    }
}
```

Relazioni principali:

```text
Corso aggrega Studente
Corso compone Lezione
```

```java
package corso.lab16;

public class Iscrizione {
    private Studente studente;
    private Corso corso;
    private String dataIscrizione;

    public Iscrizione(Studente studente, Corso corso, String dataIscrizione) {
        this.studente = studente;
        this.corso = corso;
        setDataIscrizione(dataIscrizione);
    }

    public Studente getStudente() {
        return studente;
    }

    public Corso getCorso() {
        return corso;
    }

    public String getDataIscrizione() {
        return dataIscrizione;
    }

    public void setDataIscrizione(String dataIscrizione) {
        if (dataIscrizione != null && !dataIscrizione.trim().isEmpty()) {
            this.dataIscrizione = dataIscrizione.trim();
        } else {
            this.dataIscrizione = "Data non disponibile";
        }
    }

    @Override
    public String toString() {
        return "Iscrizione{" 
                + "studente=" + studente
                + ", corso=" + corso.getCodice()
                + ", dataIscrizione='" + dataIscrizione + '\''
                + '}';
    }
}
```

`Iscrizione` rappresenta una associazione esplicita tra `Studente` e `Corso`.

La classe è utile perché la relazione ha un dato proprio: `dataIscrizione`.

## 6. File `GestioneCorsi.java`

Crea:

```text
src/corso/lab16/GestioneCorsi.java
```

Inserisci:

```java
package corso.lab16;

import java.util.ArrayList;

public class GestioneCorsi {

    public void stampaCorso(Corso corso) {
        System.out.println(corso);
    }

    public void stampaIscrizione(Iscrizione iscrizione) {
        System.out.println(iscrizione);
    }

    public Studente trovaStudentePerMatricola(ArrayList<Studente> studenti, String matricola) {
        if (studenti == null || matricola == null) {
            return null;
        }

        for (Studente studente : studenti) {
            if (studente.getMatricola().equalsIgnoreCase(matricola)) {
                return studente;
            }
        }

        return null;
    }

    public void stampaRisultatoRicerca(Studente studente) {
        if (studente == null) {
            System.out.println("Studente non trovato.");
        } else {
            System.out.println("Studente trovato: " + studente);
        }
    }
}
```

`GestioneCorsi` mostra una dipendenza.

Usa altri oggetti nei metodi, ma non li conserva come attributi.

```java
package corso.lab16;

public class MainLab16 {
    public static void main(String[] args) {
        Studente s1 = new Studente("S001", "Anna", "Rossi");
        Studente s2 = new Studente("S002", "Marco", "Verdi");
        Studente s3 = new Studente("S003", "Luca", "Bianchi");

        Corso corso = new Corso("J101", "Java Object Oriented");

        corso.aggiungiLezione("Classi e oggetti", 120);
        corso.aggiungiLezione("Incapsulamento", 150);
        corso.aggiungiLezione("Relazioni tra classi", 180);

        corso.iscriviStudente(s1);
        corso.iscriviStudente(s2);
        corso.iscriviStudente(s3);

        Iscrizione iscrizione1 = new Iscrizione(s1, corso, "2026-05-04");
        Iscrizione iscrizione2 = new Iscrizione(s2, corso, "2026-05-05");

        GestioneCorsi gestione = new GestioneCorsi();

        System.out.println("=== CORSO ===");
        gestione.stampaCorso(corso);

        System.out.println();
        System.out.println("=== LEZIONI: COMPOSIZIONE ===");
        corso.stampaLezioni();

        System.out.println();
        System.out.println("=== STUDENTI: AGGREGAZIONE ===");
        corso.stampaStudenti();

        System.out.println();
        System.out.println("=== ISCRIZIONI: ASSOCIAZIONE ESPLICITA ===");
        gestione.stampaIscrizione(iscrizione1);
        gestione.stampaIscrizione(iscrizione2);

        System.out.println();
        System.out.println("=== RICERCA: DIPENDENZA DA PARAMETRI ===");
        Studente trovato = gestione.trovaStudentePerMatricola(corso.getStudenti(), "S002");
        gestione.stampaRisultatoRicerca(trovato);

        Studente nonTrovato = gestione.trovaStudentePerMatricola(corso.getStudenti(), "S999");
        gestione.stampaRisultatoRicerca(nonTrovato);
    }
}
```

## 13. Sintesi

La sintesi è:

```text

Corso -> aggrega studenti e compone lezioni

GestioneCorsi -> dipende da altre classi per eseguire operazioni
```

L'obiettivo non è solo compilare il codice.
